\subsection{Theorem 1.9}
\begin{wts}
    If $a|(b+c)$, and $\gcd(b,c)=1$, then
    \[
    \gcd(a,b) = \gcd(a,c) = 1
    \]
\end{wts}
\begin{proof}
    Suppose $\gcd(a,b)\geq 1$, then there exists some $y\geq 2$ such that $y|a$ and $y|b$,
\end{proof}
